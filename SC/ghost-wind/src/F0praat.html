form Listing F0
	sentence File ...
endform
	Read from file... 'file$'
#  human voices are roughly in the range of 80 Hz to 1100 Hz 
#  for normal male and female voices together
	To Pitch... 0 75 1200
	Down to PitchTier
	Save as short text file... tempF0.txt
select all
Remove
