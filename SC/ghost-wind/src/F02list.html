#!/bin/bash
# convert PitchTier from praat to coll
# Created by by.cmsc@gmail.com on 11/02/12

praatfile=$1
l=`cat $praatfile |wc -l`
ll=`expr $l - 6`
tail -n $ll $praatfile > "$praatfile.tmp"
coll="$praatfile.coll"
h=`cat $praatfile.tmp |wc -l`
k=`expr $h / 2`

while read line1
do
read line2
echo "($line1 $line2)" >> $coll
done < $praatfile.tmp

rm $praatfile.tmp
